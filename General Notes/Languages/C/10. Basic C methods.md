`<stdio.h>` — Standard Input/Output

```c
int printf(const char *format, ...);         
// Prints formatted output to stdout

int fprintf(FILE *stream, const char *format, ...);  
// Prints formatted output to a file

int sprintf(char *str, const char *format, ...);     
// Writes formatted output to a string

int snprintf(char *str, size_t size, const char *format, ...); 
// Writes formatted output to string with size limit

int scanf(const char *format, ...);          
// Reads formatted input from stdin

int fscanf(FILE *stream, const char *format, ...);   
// Reads formatted input from file

int sscanf(const char *str, const char *format, ...); 
// Reads formatted input from string

int puts(const char *str);                   
// Writes a string followed by a newline

int putchar(int c);                          
// Writes a character to stdout

char *fgets(char *str, int n, FILE *stream); 
// Reads a line from a file (or stdin)

int fputs(const char *str, FILE *stream);    
// Writes a string to a file

int fputc(int c, FILE *stream);              
// Writes a character to a file

FILE *fopen(const char *filename, const char *mode); 
// Opens a file

int fclose(FILE *stream);                    
// Closes a file

size_t fread(void *ptr, size_t size, size_t count, FILE *stream); 
// Reads binary data from a file

size_t fwrite(const void *ptr, size_t size, size_t count, FILE *stream); 
// Writes binary data to a file

int feof(FILE *stream);                      
// Checks end-of-file

int fseek(FILE *stream, long offset, int whence);    
// Moves file pointer to position

long ftell(FILE *stream);                    
// Returns current file position

void rewind(FILE *stream);                   
// Rewinds file to beginning

void perror(const char *str);                
// Prints a descriptive error message

```

# `<stdlib.h>` — General Utilities

```c
void *malloc(size_t size);                   
// Allocates memory block

void *calloc(size_t num, size_t size);       
// Allocates and zeroes memory block

void *realloc(void *ptr, size_t size);       
// Resizes previously allocated memory block

void free(void *ptr);                        
// Frees allocated memory

void exit(int status);                       
// Terminates the program immediately

int system(const char *command);             
// Executes a shell command

int atoi(const char *str);                   
// Converts string to int

long int strtol(const char *str, char **endptr, int base); 
// Converts string to long with error checking

double strtod(const char *str, char **endptr); 
// Converts string to double with error checking

int rand(void);                              
// Returns a pseudo-random number

void srand(unsigned int seed);               
// Seeds the random number generator

void *bsearch(const void *key, const void *base, size_t nmemb, size_t size,
              int (*compar)(const void *, const void *)); 
// Binary search

void qsort(void *base, size_t nmemb, size_t size,
           int (*compar)(const void *, const void *)); 
// Quicksort

```

# `<string.h>` — String and Memory Operations

```c
size_t strlen(const char *str);              
// Returns length of string

char *strcpy(char *dest, const char *src);  
// Copies string (no bounds check)

char *strncpy(char *dest, const char *src, size_t n); 
// Copies up to n chars

char *strcat(char *dest, const char *src);  
// Appends src to dest

char *strncat(char *dest, const char *src, size_t n); 
// Appends up to n chars

int strcmp(const char *s1, const char *s2);  
// Compares two strings

int strncmp(const char *s1, const char *s2, size_t n); 
// Compares first n characters

char *strchr(const char *str, int c);        
// Finds first occurrence of character

char *strrchr(const char *str, int c);       
// Finds last occurrence of character

char *strstr(const char *haystack, const char *needle); 
// Finds first occurrence of substring

char *strtok(char *str, const char *delim);  
// Tokenizes string by delimiter

void *memcpy(void *dest, const void *src, size_t n); 
// Copies n bytes of memory

void *memmove(void *dest, const void *src, size_t n); 
// Safe copy for overlapping memory

void *memset(void *str, int c, size_t n);    
// Fills memory with a byte value

int memcmp(const void *s1, const void *s2, size_t n); 
// Compares n bytes of memory

```
