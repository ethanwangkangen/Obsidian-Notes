# Basics

```python
import requests
r = request.get('https://example.com')
r = requests.post('https://httpbin.org/post', data={'key': 'value'})
r = requests.put('https://httpbin.org/put', data={'key': 'value'})
r = requests.delete('https://httpbin.org/delete')
r = requests.head('https://httpbin.org/get')
r = requests.options('https://httpbin.org/get')

# Passing parameters
payload = {'key1': 'value1', 'key2': 'value2'}
r = requests.get('https://httpbin.org/get', params=payload)
print (r.url)  # https://httpbin.org/get?key2=value2&key1=value1

# Reading Response Content
print (r.text)    # '[{"repository":{"open_issues":0,"url":"https://github.com/...
print (r.content) # b'[{"repository":{"open_issues":0,"url":"https://github.com/...
print (r.json)    # [{'repository': {'open_issues': 0, 'url': 'https://github.com/...

# Raw response content
r = requests.get('https://api.github.com/events', stream=True)
r.raw             # <urllib3.response.HTTPResponse object at 0x101194810>
r.raw.read(10)    # b'\x1f\x8b\x08\x00\x00\x00\x00\x00\x00\x03'

# HTTP Headers
url = 'https://api.github.com/some/endpoint'
headers = {'user-agent': 'my-app/0.0.1'}
r = requests.get(url, headers=headers)

# POST requests with form data
payload = {'key1': 'value1', 'key2': 'value2'}

payload_tuples = [('key1', 'value1'), ('key1', 'value2')] # Multiple values for each key: tuple
payload_dict = {'key1': ['value1', 'value2']}             # Multiple values for each key: dict

r = requests.post('https://httpbin.org/post', data=payload)

# POST as json
payload = {'some': 'data'}
r = requests.post(url, json=payload)

# POST as Multipart-Encoded file
url = 'https://httpbin.org/post'
files = {'file': open('report.xls', 'rb')}
files = {'file': ('report.xls', open('report.xls', 'rb'), 'application/vnd.ms-excel', {'Expires': '0'})} # Can se filename, content_type, headers explicitly
files = {'file': ('report.csv', 'some,data,to,send\nanother,row,to,send\n')} # Send strings to be received as file

r = requests.post(url, files=files)

# Response Status codes
r.status_code
print(r.status_code == requests.codes.ok)

# Response header
r.headers # Is a dictionary. {'content-encoding':'gzip', ... }
r.headers['Content-Type']

# Getting cookies
r = requests.get(url)
r.cookies['example_cookie_name']

# Sending cookies
cookies = {"cookiekey": "cookieval"}
r = requests.get(url, cookies=cookies)

# Cookies being returned
jar = requests.cookies.RequestsCookieJar()
jar.set('tasty_cookie', 'yum', domain='httpbin.org', path='/cookies')
jar.set('gross_cookie', 'blech', domain='httpbin.org', path='/elsewhere')
url = 'https://httpbin.org/cookies'
r = requests.get(url, cookies=jar)
r.text

```
