```python
import argparse

# Builds the parser, adds arguments, returns it
def build_parser() -> argparse.ArgumentParser:

    # The stuff here shows when calling -h
    p = argparse.ArgumentParser(
        description="HTTP probe",
        epilog="Example:\n  probe example.com --https -t 3",
        formatter_class=argparse.RawDescriptionHelpFormatter,  # preserves newlines in epilog
    )
    
    # 3 types of argument kinds
    
    # 1. Positional
    p.add_argument("host", help="Hostname or IP")
    
    # 2. Optional with value
    p.add_argument("-p", "--port", type=int, default=None, help="Port (default 80/443)")
    
    # 3. Optional with boolean
    p.add_argument("--https", action="store_true", help="Use HTTPS")
    p.add_argument("-t", "--timeout", type=float, default=5.0, help="Timeout in seconds")
    p.add_argument("-v", "--verbose", action="count", default=0, help="Increase verbosity (-vv)")
    return p

def main(argv=None):
    parser = build_parser()
    args = parser.parse_args(argv)   # pass argv in tests; None â†’ sys.argv[1:]
    # use args.host, args.port, args.https, args.timeout, args.verbose

```


So 
1. Build the parser `p = argparse.ArgumentParser(..)`
2. Add arguments `p.add_argument()`
3. Parse the arguments and store`arg = p.parse_args(argv)`
4. Use the arguments with `args.<argument name>

## Action
action types:
- `store`: save the value once
- `store_true/store_false`: booleans for flags
- `append`: collect repeated flags into a list
- `count`: count occurrences (`---v` -->3)
