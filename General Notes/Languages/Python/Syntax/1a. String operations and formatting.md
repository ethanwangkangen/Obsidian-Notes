# String operations
```python
s = "GET /admin HTTP/1.1"
s[0]        # 'G'
s[-1]       # last char '1'
s[4:9]      # substring ' /adm' (start inclusive, stop exclusive)
s[:3]       # first 3 chars
s[3:]       # from index 3 to end
s[::2]      # every 2nd char
s[::-1]     # reversed string

s.lower()                 # lowercase
s.upper()                 # UPPERCASE
s.capitalize()            # First char uppercase
s.title()                 # Title Case
s.strip()                 # remove leading/trailing whitespace
s.lstrip(), s.rstrip()    # left/right only

s.split(sep=None, maxsplit=-1)   # split on whitespace or sep
" a b ".split()  # ['a','b']
"1,2,3".split(",", 1)  # ['1', '2,3']

" ".join(["a","b"])     # 'a b'  â€” prefer join for concatenation of many pieces

s.replace("old","new", count=-1)  # replace occurrences

s.find(sub)              # index of first occurrence or -1
s.index(sub)             # same, but raises ValueError if not found
s.rfind(sub)             # last occurrence
s.startswith(prefix), s.endswith(suffix)
"token" in s             # membership test (case-sensitive)
```
