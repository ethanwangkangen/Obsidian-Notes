
## Visitor 
- Instead of type based branching
```cpp
if (auto c = dynamic_cast<Circle*>(shape)) { ... }
else if (auto s = dynamic_cast<Square*>(shape)) { ... }
```
- We want a way to have dynamic resolution
	- But virtual functions in Shape will clutter Shape, especially if there are a lot of operations
	- Solutions: use a Visitor

```cpp
struct Circle;  
struct Square;  
  
struct ShapeVisitor {  
	virtual void visit(Circle&) = 0;  
	virtual void visit(Square&) = 0;  
	virtual ~ShapeVisitor() = default;  
};

struct Shape {  
	virtual void accept(ShapeVisitor&) = 0;  
	virtual ~Shape() = default;  
};

struct Circle : Shape {
    void accept(ShapeVisitor& v) override {
        v.visit(*this);
    }
};

struct Square : Shape {
    void accept(ShapeVisitor& v) override {
        v.visit(*this);
    }
};

struct RenderVisitor : ShapeVisitor {
    void visit(Circle& c) override {
        // render circle
    }

    void visit(Square& s) override {
        // render square
    }
};

int main() {
	Shape* s = getShape(); // might point to a Circle or a Square object
	RenderVisitor renderer;

	s->accept(renderer);
}
```
- `Shape* s` either points to Circle or Square object
- `s->accept(renderer)` 
	- Virtual dispatch happens here.
	- vtable chooses `Circle::accept`, for eg. if s points to Circle
- `Circle::accept()` calls `v.visit(*this)`, which chooses the overload that takes in a Circle reference.

- **Double dispatch** paradigm

|Mechanism|Type resolved|When|
|---|---|---|
|virtual call (`accept`)|Shape type|runtime|
|overload (`visit`)|Visitor function|compile time|
