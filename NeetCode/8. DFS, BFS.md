# DFS
> Difference from BFS: call blindly with recursive, push blindly with iterative. Don't do check for validity until within the function!
> VS BFS: validity check happens **before** even pushing.

- Check visited
- Visit
- Process
- DFS neighours
```cpp
void dfs(int node, unordered_map<int, vector<int>>& graph, unordered_set<int>& visited) {
	// Check visited
    if (visited.count(node)) return;
	
	// Visit
    visited.insert(node);

    // Process node
    // e.g. cout << node;

	// Dfs neighbours
    for (int neighbour : graph[node]) {
        dfs(neighbour, graph, visited);
    }
}
```


# BFS
- Push root into queue and visit
- Loop
	- Pop front then process
	- For each neighbour, if not visitsed, push and visit
```cpp
void bfs(int start, unordered_map<int, vector<int>>& graph) {
    unordered_set<int> visited;
    queue<int> q;

	// Visit root and push into queue
    visited.insert(start);
    q.push(start);

    while (!q.empty()) {
	    // Pop front
        int node = q.front();
        q.pop();

        // Process front
        // ...

        for (int neighbour : graph[node]) {
	        // If not visited
            if (!visited.count(neighbour)) {
	            // Visit neighbour and push into queue
                visited.insert(neighbour);
                q.push(neighbour);
            }
        }
    }
}
```

## BFS level traversal
```cpp
void bfs(int start, unordered_map<int, vector<int>>& graph) {
    unordered_set<int> visited;
    queue<int> q;
    
	// Visit root and push into queue
    visited.insert(start);
    q.push(start);

    int level = 0;

    while (!q.empty()) {
	    // Check level size
        int size = q.size();
		
		// Loop for level size
        for (int i = 0; i < size; i++) {
	        // Pop front
            int node = q.front();
            q.pop();

            // process node at current level

            for (int neighbour : graph[node]) {
	            // If not visited
                if (!visited.count(neighbour)) {
	                // Visit neighbour and push into queue
                    visited.insert(neighbour);
                    q.push(neighbour);
                }
            }
        }

        level++; // move to next layer
    }
}
```

## BFS on GRID
- Only differences:
	- Visited 2D vector
	- Before pushing neighbours, check if not visited yet **and within bounds**
```cpp
vector<pair<int,int>> directions = {{1,0}, {-1,0}, {0,1}, {0,-1}};

void bfs(vector<vector<int>>& grid, int r, int c) {
    int rows = grid.size(), cols = grid[0].size();

    queue<pair<int,int>> q;
    
	// Take note of visited 2d vector!
    vector<vector<bool>> visited(rows, vector<bool>(cols, false));
    
    q.push({r, c});
    visited[r][c] = true;

    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();

        // process cell

        for (auto [dx, dy] : directions) {
            int nx = x + dx;
            int ny = y + dy;

            if (nx >= 0 && ny >= 0 && nx < rows && ny < cols &&
                !visited[nx][ny]) {

                visited[nx][ny] = true;
                q.push({nx, ny});
            }
        }
    }
}
```
