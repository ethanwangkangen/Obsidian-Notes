- IMPORTANT EXAMPLE!
- Instead of comparing hash tables each time, have a count of **unique** words, then each `required[added]==seen[added]`, have ++, and each time we delete a word right as `required[added]==seen[added]`, have--.
	- Then just check when have == count, where count is count of **unique** words
- ALSO remember that when we check for presence of something in a map, with `mp[obj]`, it sets `mp[obj]` to 0!!!!
	- If want to just check for presence, use `.count()`


```cpp
class Solution {
public:
    vector<int> findSubstring(string s, vector<string>& words) {
        vector<int> res;

        const int wordLength = words[0].size();
        const int totalLength = wordLength * words.size();
         // unique words

        unordered_map<std::string_view, int> required;
        for (auto& word: words) required[word]++;
        const int requiredWords = required.size();

        for (int i{}; i < wordLength; ++i) {
            int l = i;
            int have{};
            unordered_map<std::string_view, int> seen{};
            for (int r {l+wordLength}; r<= s.size(); r += wordLength) {
                // add the next chunk
                std::string_view added (s.data() + r-wordLength, wordLength); 
                seen[added] +=1;
                if (!required.count(added)) { //reset window
                    l = r;
                    seen.clear();
                    have=0;
                    continue;
                }

                if (seen[added] == required[added]) have+=1;

                // increment l until valid
                std::string_view possibleRemove (s.data() + l, wordLength);
                if (r-l > totalLength) {
                    if (seen[possibleRemove] == required[possibleRemove]) have--;
                    seen[possibleRemove]--;
                    l+= wordLength;
                }
                if (have==requiredWords) res.push_back(l);
            }


        }
        return res;

    }
};
```
