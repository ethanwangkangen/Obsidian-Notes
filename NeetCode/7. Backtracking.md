# Pick or skip
Formula
- Check if need to add to result
- Prune
- Take branch
- Undo decision
- Don't take branch

```cpp
class Solution {
public:
    void dfs(int n, int k, vector<vector<int>>& res, vector<int>& state, int bound) {
        // check if add to result
        if (state.size() == k) {
            res.push_back(state);
            return;
        }

        // prune
        if (state.size() > k or bound > n) return;

        // take
        state.push_back(bound);
        dfs(n, k, res, state, bound+1);

        // undo decision
        state.pop_back();
        
        // don't take
        dfs(n, k, res, state, bound+1);

    }
    vector<vector<int>> combine(int n, int k) {
        // n is biggest number (inclusive), k is size of combination    
        vector<vector<int>> res;
        vector<int> initState {};
        dfs(n, k, res, initState, 1);
        return res;
    }
};
```

For avoiding duplicates:
- Check
- Prune
- Take -> push back
- Rollback
- Don't take -> DON'T even pushback! Just find the next bound and dfs().

- Can also do it the 'permutations way', where **every** node is valid. see the permutations secion below
```cpp
void dfs(vector<vector<int>>& res, vector<int>& nums, int bound, vector<int>& state) {
    // check: NOTE -> ONLY THE LEAVES ARE VALID HERE!
    if (bound == nums.size()) {
        res.push_back(state);
        return;
    }

    // prune
    if (bound >= nums.size()) return;

    // take
    int elem {nums[bound]};
    state.push_back(elem);
    dfs(res, nums, bound+1, state);

    // rollback
    state.pop_back();

    // don't take
    while (nums[bound] == elem and bounds < nums.size()) bound++;
    dfs(res, nums, bound, state);
}
```

# Permutations
- Choose from remaining
```cpp
void dfs(vector<int>& nums, vector<bool>& used, vector<int>& path, vector<vector<int>>& res) {
    if (path.size() == nums.size()) {
        res.push_back(path);
        return;
    }

    for (int i = 0; i < nums.size(); i++) {
        if (used[i]) continue;

        used[i] = true;
        path.push_back(nums[i]);

        dfs(nums, used, path, res);

        path.pop_back();
        used[i] = false;
    }
}
```

## Combinations with duplicates can fall under the 'choose from remaining' template.
```cpp
void dfs(int bound, vector<int>& nums, vector<int>& state, vector<vector<int>>& res) {
    res.push_back(state);  // every node is a valid subset

    for (int i = bound; i < nums.size(); i++) {
        // skip duplicates at the same level
        if (i > bound && nums[i] == nums[i - 1]) continue;
        state.push_back(nums[i]);         // pick nums[i]
        dfs(i + 1, nums, state, res);    // recurse
        state.pop_back();                // backtrack
    }
}
```
